#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
const int MAXM = 20 + 5;

int n, m;
string s;
int pos[MAXM];
int dp[1 << MAXM];

// Độ chậm của một hoán vị
int slow(int mask) {
    int last = -1, res = 0;
    for (int i = 0; i < m; ++i) {
        if (mask & (1 << i)) {
            if (last != -1) {
                res += abs(pos[i] - pos[last]);
            }
            last = i;
        }
    }
    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    cin >> n >> m >> s;
    for (int i = 0; i < m; ++i) {
        char c;
        cin >> c;
        pos[c - 'a'] = i;
    }

    memset(dp, 0x3f, sizeof(dp));
    dp[0] = 0;

    // Tìm tất cả các hoán vị của m chữ cái Latin đầu tiên
    for (int mask = 0; mask < (1 << m); ++mask) {
        dp[mask] = slow(mask);
    }

    // Tìm độ chậm nhỏ nhất
    for (int mask = 0; mask < (1 << m); ++mask) {
        for (int submask = mask; submask > 0; submask = (submask - 1) & mask) {
            dp[mask] = min(dp[mask], dp[submask] + dp[mask ^ submask]);
        }
    }

    cout << dp[(1 << m) - 1] << '\n';

    return 0;
}
